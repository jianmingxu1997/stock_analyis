### 2. 不符条件列生成

**位置**: `generate-dashboard.js` 第 73-95 行

```javascript
function generateConditionColumn(row, config) {
    const failedConditions = [];
    const conditionNames = {
        '日线金叉': '日线金叉',
        '当日收红': '当日收红',
        'RSI<70': 'RSI<70',
        '量比>1': '量比>1',
        '涨幅<5%': '涨幅<5%',
        '14 日>-7%': '14 日>-7%',
        '20 日>-10%': '20 日>-10%',
        'MA20 向上': 'MA20 向上',
        '周线金叉': '周线金叉',
        '3 月>3%': '3 月>3%'
    };
    
    config.conditionColumns.forEach(col => {
        if (row[col] === '✗') {
            failedConditions.push(conditionNames[col] || col);
        }
    });
    
    return failedConditions.length > 0 ? failedConditions.join(';') : '-';
}
```

**要点**:
- 读取 Excel 中打✗的列
- 多个条件用分号连接
- 全对显示"-"

### 5. 滚轮缩放逻辑

**位置**: `generate-dashboard.js` 第 665-695 行

```javascript
chart.getZr().on('mousewheel', function(params) {
    const delta = params.wheelDelta;
    let start = dataZoom.start || 0;
    const end = 100;  // 右侧始终固定
    
    // 向上滚动：左侧右移，缩小范围
    // 向下滚动：左侧左移，扩大范围
    const step = delta > 0 ? 10 : -10;  // 每次 10%
    start = Math.max(0, Math.min(90, start + step));
    
    chart.setOption({ 
        dataZoom: [
            { start: start, end: 100 }, 
            { start: start, end: 100 } 
        ] 
    });
    params.event.preventDefault();
});

// 窗口大小自适应
window.addEventListener('resize', function() {
    chart.resize();
});
```

**关键配置**:
- `zoomOnMouseWheel: false` - 禁用 ECharts 默认滚轮行为
- `end: 100` - 右侧始终固定在最新一天
- `step: 10` - 每次滚动 10% 范围
- `resize()` - 窗口大小变化时自动调整图表

let showOnlyScore10 = false;

function filterTable() {
    const tr = table.getElementsByTagName('tr');
    for (let i = 1; i < tr.length; i++) {
        const tds = tr[i].getElementsByTagName('td');
        const industryTd = tds[0];  // 行业列
        const scoreTd = tds[6];     // 得分列（第 7 列）
        const score = parseFloat(scoreTd.textContent);
        
        let showRow = true;
        
        // 行业筛选
        if (filter && industryTd.textContent.indexOf(filter) === -1) 
            showRow = false;
        
        // 10 分筛选
        if (showOnlyScore10 && score !== 10) 
            showRow = false;
        
        tr[i].style.display = showRow ? '' : 'none';
    }
}

## 🐛 已知问题/注意事项

1. **中文字符编码**: 保存时必须使用 UTF-8
2. **得分列索引**: filterTable 中固定为 `tds[6]`，如修改列顺序需同步更新
3. **周线聚合**: 使用 ISO 周数，可能与自然周有差异
4. **数据更新**: 更新 Excel 后需重新运行脚本生成 HTML
5. **文件大小**: 含 Logo 的 HTML 约 9MB，分享时注意平台限制
